import {Component, OnInit} from '@angular/core';
import {JsonService} from "../../../shared/services/json/json.service";
import {SubscriptionsService} from "../../../shared/services/subscriptions/subscriptions.service";
import {Pick, NewPick} from "../../../shared/interfaces/pick";
import {BetSlipService} from "../../../shared/services/bet-slip/bet-slip.service";
import {Subscription} from "rxjs";

@Component({
  selector: 'workspace-in-play',
  templateUrl: './in-play.component.html',
  styleUrls: ['./in-play.component.scss']
})
export class InPlayComponent implements OnInit {

  activeEl: string;
  inPlayTable: any;
  elementsObj: any;
  betSlipCartChanged: Subscription;

  constructor(private jsonService: JsonService,
              public subscriptionsService: SubscriptionsService,
              private betSlipService: BetSlipService) {
    this.elementsObj = {
      soccer: "Soccer",
      basketball: 'Basketball',
      tennis: 'Tennis',
      formula_1: 'Formula 1',
      snooker: 'Snooker',
      motorbike: 'Motorbike'

    }

    this.betSlipCartChanged = this.subscriptionsService.getBetSlipChanges().subscribe(data => {
      if (data) {
        this.inPlayTable = this.betSlipService.checkPickIfIsInCart(this.inPlayTable, JSON.parse(localStorage.getItem('betSlip_cart')))
      }
    })

  }

  ngOnInit(): void {
    this.activeEl = 'soccer';
    this.jsonService.getJson(`in-play-${this.activeEl}`).subscribe(data => {
      this.inPlayTable = this.betSlipService.checkPickIfIsInCart(data, JSON.parse(localStorage.getItem('betSlip_cart')))
      console.log(data)
    })
  }

  getInPlays(sportName) {
    this.activeEl = sportName;
    this.jsonService.getJson(`in-play-${sportName}`).subscribe(data => {
      this.inPlayTable = this.betSlipService.checkPickIfIsInCart(data, JSON.parse(localStorage.getItem('betSlip_cart')))
    })
  }

  addBetToCart(e, pick) {
    if (pick.inCart === true) {
      pick.inCart = false;
      this.betSlipService.removePickFromBetSlip(pick.id);
    } else {
      pick.inCart = true;
      const objectToPush: NewPick = {
        live: pick.live,
        event_id: pick.id,
        event_name: pick.event_name,
        finished: pick.finished,
        locked: pick.locked,
        sport_name: pick.sport_name,
        option_name: pick.option_name,
        option_id: pick.id,
        stake: pick.inPlayPickMultiplier,
        price: 10,
        selected: true
      }
      this.betSlipService.addPickToBetSlip(objectToPush);
    }
  }
}
