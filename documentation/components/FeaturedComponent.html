import {Component, OnInit} from '@angular/core';
import {JsonService} from "../../../shared/services/json/json.service";
import {OwlOptions} from "ngx-owl-carousel-o";
import {SubscriptionsService} from "../../../shared/services/subscriptions/subscriptions.service";
import {Pick} from '../../../shared/interfaces/pick'

@Component({
  selector: 'workspace-featured',
  templateUrl: './featured.component.html',
  styleUrls: ['./featured.component.scss']
})
export class FeaturedComponent implements OnInit {

  featuredArray: any;
  pickInterface: Pick;
  cartFromLocalStorage: any;
  featuredArrayLenght: any;
  featuredCarouselOptions: OwlOptions = {
    loop: true,
    dots: false,
    nav: true,
    navText: ['', ''],
    autoplay: true,
    autoplayTimeout: 10000,
    autoplayHoverPause: true,
    autoplayMouseleaveTimeout: 2000,
    responsive: {
      0: {
        items: 1
      },
      550: {
        items: 2
      },
      740: {
        items: 3
      }

    }
  }

  constructor(private jsonService: JsonService, public subscriptionsService: SubscriptionsService) {
    this.featuredArrayLenght = 0;
  }

  ngOnInit(): void {
    this.cartFromLocalStorage = JSON.parse(localStorage.getItem('100BetCart'))
    this.jsonService.getJson('featured-matches').subscribe(e => {
      this.featuredArray = this.makeChosenOrNot(e, this.cartFromLocalStorage);
      this.featuredArrayLenght = this.featuredArray.length;
    })

  }

  addBetToCart(obj, pick) {
    if (pick.inCart === true) {
      pick.inCart = false;
      this.subscriptionsService.triggerRemoveFromCart(pick.id);
    } else {
      pick.inCart = true;
      this.subscriptionsService.triggerAddPickToCart({
        pick: this.pickInterface = {
          id: pick.id,
          date: obj.featuredDate,
          hour: obj.featuredTime,
          multiplier: pick.featuredPickMultiplier,
          teams: [
            {
              team: obj.featuredTeams[0].featuredTeamName
            },
            {
              team: obj.featuredTeams[1].featuredTeamName
            }
          ]
        }
      })
    }
  }

  makeChosenOrNot(firstArr, secondArr) {
    const tempArr1 = secondArr.map((e) => e.id)
    firstArr.forEach((e) => {
      e.featuredPicks.filter((f) => {
        if (tempArr1.includes(f.id)) {
          f.inCart = true;
        }
      })
    })
    return firstArr;
  }

}
