import {Component, OnInit} from '@angular/core';
import {JsonService} from "../../../shared/services/json/json.service";

@Component({
  selector: 'workspace-upcoming',
  templateUrl: './upcoming.component.html',
  styleUrls: ['./upcoming.component.scss']
})
export class UpcomingComponent implements OnInit {

  activeEl: string;
  upcomingTable: any;
  elementsObj: any;
  defaoultOption = {
    optionName: 'Asian Handicap & Goal Line',
    optionDisplayName: 'Asian Handicap & Goal Line',
    optionValue: '129q8wry283b7bc'
  }


  selectOptions = [
    {
      optionName: 'Asian Handicap & Goal Line',

    {
      optionName: 'Handicap 0:1',
      optionDisplayName: 'Handicap 0:1',
      optionValue: 'e23fdv'
    },
    {
      optionName: 'Totals 2nd Half',
      optionDisplayName: 'Totals 2nd Half',
      optionValue: '23fe'
    }
  ]

  constructor(private jsonService: JsonService) {
    this.elementsObj = {
      soccer: "Soccer",
      basketball: 'Basketball',
      tennis: 'Tennis',
      formula_1: 'Formula 1',
      snooker: 'Snooker',
      motorbike: 'Motorbike'
    }
  }

  ngOnInit(): void {
    this.activeEl = 'soccer';
    this.jsonService.getJson(`upcoming-${this.activeEl}`).subscribe(data => {
      this.upcomingTable = data;
    })
  }

  getUpcoming(sportName) {
    this.activeEl = sportName;
    this.jsonService.getJson(`upcoming-${sportName}`).subscribe(data => {
      this.upcomingTable = data;
    })
  }

  test(e){
    console.log(e);
  }

}
