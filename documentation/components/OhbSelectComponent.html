import {Component, Input, OnInit, Output, EventEmitter} from '@angular/core';

@Component({
  selector: 'workspace-ohb-select',
  templateUrl: './ohb-select.component.html',
  styleUrls: ['./ohb-select.component.scss']
})
export class OhbSelectComponent implements OnInit {
  @Output() changeValue: EventEmitter<any> = new EventEmitter<any>();

  state: boolean;
  @Input() options: any;
  @Input() selected: any;

  constructor() {
  }

  ngOnInit(): void {
  }

  checkState() {
    this.state = !this.state;
  }

  clickedOutside() {
    this.state = false;
  }

  selectOption(value){
    this.changeValue.emit(value);
    this.state = false;
    this.selected = value;
  }
}

<div class="ohb-select" (clickOutside)="clickedOutside()">
  <div class="ohb-placeholder" (click)="checkState()" [ngClass]="{'opened': state}">
    <span [innerHTML]="selected.optionDisplayName"></span>
    <i class="fas fa-angle-down"></i>
  </div>
  <div class="ohb-options">
    <ul [ngClass]="{'open': state}" #selectList>
      <li *ngFor="let option of options" [value]="option.optionValue" (click)="selectOption(option)">
        <span [innerHTML]="option.optionDisplayName"></span>
      </li>
    </ul>
  </div>
</div>
